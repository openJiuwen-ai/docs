本章节演示了如何基于openJiuwen快速构建大模型应用。通过本示例，你将会了解到如下信息：

- 如何使用openJiuwen开发大模型应用。
- 如何使用`ModelFactory`进行模型调用。
- 如何给大模型应用添加记忆功能。

# 前提条件

Python的版本应高于或者等于Python 3.11版本，建议使用3.11.4版本，使用前请检查Python版本信息。

# 初始化大模型调用

通过`create_model_config`方法配置模型提供商（`MODEL_PROVIDER`）、模型名称（`MODEL_NAME`）、模型请求的URL地址（`API_BASE`）和鉴权Token（`API_KEY`）等模型基本信息，示例代码如下：

```python
import os
from openjiuwen.core.component.common.configs.model_config import ModelConfig
from openjiuwen.core.utils.llm.base import BaseModelInfo

# 从环境变量中取得大模型相关参数信息
API_BASE = os.getenv("API_BASE", "")
API_KEY = os.getenv("API_KEY", "")

MODEL_NAME = os.getenv("MODEL_NAME", "")
MODEL_PROVIDER = os.getenv("MODEL_PROVIDER", "")

def _create_model_config() -> ModelConfig:
    return ModelConfig(
        model_provider=MODEL_PROVIDER,
        model_info=BaseModelInfo(
            model=MODEL_NAME,
            api_base=API_BASE,
            api_key=API_KEY,
            temperature=0.7,
            top_p=0.9,
            timeout=30,
        ),
    )

# 配置大模型相关参数信息
model_config = _create_model_config()
```

使用`ModelFactory`的`get_model`方法获取大模型实例，示例代码如下：

```python
# 使用ModelFactory获取大模型实例
self._llm = ModelFactory().get_model(model_provider=self.model_config.model_provider, 
                                     api_base=self.model_config.model_info.api_base, 
                                     api_key=self.model_config.model_info.api_key)
```

# 创建Agent

本示例展示了一个带会话记忆的聊天Agent，可继承上一轮的结论。该类继承自`BaseAgent`类，在`init`方法里，接收`AgentConfig`，初始化父类，并通过`ModelFactory`获取大模型实例。
`invoke`方法用于调用大模型。调用前先将用户输入封装为`BaseMessage`写入`self.chat_history`，再以完整的`self.chat_history`作为模型输入；为便于下一轮延续语境，会在得到回复后把本次模型输出追加回`self.chat_history`。
示例代码如下：

```python
# 定义自己的智能体类
class ChatBot(BaseAgent):
    def __init__(self, agent_config : AgentConfig):
        super().__init__(agent_config)
        self.model_config = _create_model_config()
        # 使用ModelFactory调用大模型
        self._llm = ModelFactory().get_model(model_provider=self.model_config.model_provider, 
                                             api_base=self.model_config.model_info.api_base, 
                                             api_key=self.model_config.model_info.api_key)
        # self.chat_history存储历史对话消息，self.max_history_num为存储历史对话消息的长度
        self.chat_history = []
        self.max_history_num = 1

    async def invoke(self, inputs: Dict, runtime: Runtime = None):
        if len(self.chat_history) > self.max_history_num:
            # 如果历史对话消息的长度大于self.max_history_num，则把最旧的信息从历史对话消息中删除
            self.chat_history.pop(0)
        # inputs是智能体的输入，以llm_inputs为键，通过BaseMessage组合为当前用户输入信息
        self.chat_history.append(BaseMessage(role="user", content=inputs.get("llm_inputs")).model_dump(exclude_none=True))
        messages = [
            SystemMessage(content="你是一个AI助手").model_dump(exclude_none=True),
        ]
        messages.extend(self.chat_history)
        # 首次调用时，self.chat_history长度为1，存放当前用户输入信息，后续调用时，self.chat_history长度为2，存放上次模型结果的输出和当前用户输入信息，从而实现Agent的记忆功能。调用模型的ainvoke方法得到模型的输出。
        res = await self._llm.ainvoke(model_name=self.model_config.model_info.model_name, messages=messages)
        # 将模型输出加到self.chat_history
        self.chat_history.append(BaseMessage(role=res.role, content=res.content).model_dump(exclude_none=True))
        return res.content

    async def stream(self, inputs: Dict, runtime: Runtime = None) -> AsyncIterator[Any]:
        # 用户可根据需要实现流式输出方法
        pass

# 实例化自己的智能体类
chat_bot = ChatBot(AgentConfig())
```

# 运行Agent

创建完成自己的Agent之后，可以通过调用Agent的`invoke`方法运行Agent，示例代码如下：

```python
inputs = {"llm_inputs": "写一个陈述句"}
res = await chat_bot.invoke(inputs)
```

# 完整示例代码

完整的示例参考如下所示：

```python
import os
import asyncio
from typing import Any, Dict, AsyncIterator

from openjiuwen.agent.config.base import AgentConfig
from openjiuwen.core.agent.agent import BaseAgent
from openjiuwen.core.component.common.configs.model_config import ModelConfig
from openjiuwen.core.runtime.runtime import Runtime
from openjiuwen.core.utils.llm.base import BaseModelInfo
from openjiuwen.core.utils.llm.messages import BaseMessage, SystemMessage
from openjiuwen.core.utils.llm.model_utils.model_factory import ModelFactory

# 从环境变量中取得大模型相关参数信息
API_BASE = os.getenv("API_BASE", "")
API_KEY = os.getenv("API_KEY", "")
MODEL_NAME = os.getenv("MODEL_NAME", "")
MODEL_PROVIDER = os.getenv("MODEL_PROVIDER", "")

def _create_model_config() -> ModelConfig:
    return ModelConfig(
        model_provider=MODEL_PROVIDER,
        model_info=BaseModelInfo(
            model=MODEL_NAME,
            api_base=API_BASE,
            api_key=API_KEY,
            temperature=0.7,
            top_p=0.9,
            timeout=30,
        ),
    )
# 定义自己的智能体类
class ChatBot(BaseAgent):
    def __init__(self, agent_config : AgentConfig):
        super().__init__(agent_config)
        self.model_config = _create_model_config()

        self._llm = ModelFactory().get_model(model_provider=self.model_config.model_provider, 
                                             api_base=self.model_config.model_info.api_base, 
                                             api_key=self.model_config.model_info.api_key)
        # self.chat_history存储历史对话消息，self.max_history_num为存储历史对话消息的长度
        self.chat_history = []
        self.max_history_num = 1

    async def invoke(self, inputs: Dict, runtime: Runtime = None):
        if len(self.chat_history) > self.max_history_num:
            # 如果历史对话消息的长度大于self.max_history_num，则把最旧的信息从历史对话消息中删除

        # inputs是智能体的输入，以llm_inputs为键，通过BaseMessage组合为当前用户输入信息

        messages = [
            SystemMessage(content="你是一个AI助手").model_dump(exclude_none=True),
        ]
        messages.extend(self.chat_history)
        # 首次调用时，self.chat_history长度为1，存放当前用户输入信息，后续调用时，self.chat_history长度为2，存放上次模型结果的输出和当前用户输入信息，从而实现Agent的记忆功能。调用模型的ainvoke方法得到模型的输出。
        res = await self._llm.ainvoke(model_name=self.model_config.model_info.model_name, messages=messages)
        # 将模型输出加到self.chat_history
        self.chat_history.append(BaseMessage(role=res.role, content=res.content).model_dump(exclude_none=True))
        return res.content

    async def stream(self, inputs: Dict, runtime: Runtime = None) -> AsyncIterator[Any]:
        # 用户可根据需要实现流式输出方法
        pass

# 实例化自己的智能体类
chat_bot = ChatBot(AgentConfig())

async def main():
    inputs = {"llm_inputs": "写一个陈述句"}
    res = await chat_bot.invoke(inputs)
    print(res)
    inputs = {"llm_inputs": "把这个陈述句改成疑问句"}
    res = await chat_bot.invoke(inputs)
    print(res)

if __name__ == "__main__":
    asyncio.run(main())
```

可以看到先输出了一个陈述句，又输出了这个陈述句修改成的疑问句，可能的执行结果如下（输出内容取决于所用的模型及其配置）：

```
今天天气晴朗，适合外出散步。
今天天气晴朗，适合外出散步吗？
```
